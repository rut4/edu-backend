v Задача со звёздочкой. Итерация по объекту коллекции с помощью метода foreach()
    public function testIterableWithForeachFunction() {
        $collection = new ReviewCollection([new Review(['name' => 'foo']), new Review(['name' => 'bar']) ]);
        $expected = array(0=>'foo', 1=>'bar');
        foreach ($collection as $_key => $review) {
            $this->assertEquals($expected[$_key], $review->getName());
        }
    }
